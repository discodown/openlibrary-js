const fetch = require("node-fetch")

class AuthorsAPI {
    /**
     * Implements the Authors API for getting information about an author and
     * their works.
     * @constructor
     * @param {int} limit - The maximimum number of results to return per request. Default: 50
     */
    constructor(limit = 50) {
        this.limit = limit;
        this.BASE_API_URL = "https://openlibrary.org/authors/"
    }

    /**
     * Get information about an author using their OpenLibrary author ID.
     * @param {string} authorId - Author's OpenLibrary author ID.
     * @returns {Promise} Promise contaiing the JSON representation of author data.
     */
    async getAuthor(authorId) {
        const req_url = `${this.BASE_API_URL}/${authorId}.json`;

        const res = await fetch(req_url);
        const data = await res.json();

        return data;
    }
    /**
     * Get a list of works by an author.
     * @param {string} authorId - Author's OpenLibrary author ID.
     * @param {int} offset - Number of entries in the results to skip. Default: 0
     * @returns {Promise} Promise contaiing the JSON representation the author's works.
     */
    async getAuthorWorks(authorId, offset = 0) {
        let req_url = `${this.BASE_API_URL}/${authorId}/works.json?limit=${this.limit}`;
        if (offset > 0) {
            req_url = req_url + `?offset=${offset}`;
        }

        const res = await fetch(req_url);
        const data = await res.json();

        return data;
    }
}

module.exports = { AuthorsAPI };
