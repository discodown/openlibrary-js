const fetch = require("node-fetch");

class SubjectsAPI {
    /**
     * Implements the Subjects API for fetching books by subject name.
     * @constructor
     * @param {Boolean} details - Whether to include extra details in the results. Default: false
     * @param {Boolean} ebooks - Whether to limit results to items with ebooks available on archive.org. Default: false
     * @param {int} limit - Maximum number of results to return. 
     */
    constructor(details = false, ebooks = false, limit = 50) {
        this.BASE_API_URL = "https://openlibrary.org/subjects/";
        this.details = details;
        this.ebooks = ebooks;
        this.limit = limit;
    }

    /**
     * Get a list of works on a subject.
     * @param {String} subject - Name of subject.
     * @param {int} offset - Number of items to skip when returning results.
     * @param {String} publishedIn - Optional year range to limit results to (ex. 1500-1600). Default: null
     * @returns {Promise} Promise object containing JSON representation of results.
     */
    async getSubjectWorks(subject, offset, publishedIn = null) {
        let req_url = `${this.BASE_API_URL}${subject}.json?limit=${this.limit}&offset=${offset}`;

        if (this.ebooks == true) {
            req_url = req_url + "&ebooks=true";
        }

        if (this.details == true) {
            req_url = req_url + "&details=true";
        }

        if (publishedIn != null) {
            req_url = req_url.concat(`&published_in=${publishedIn}`);
        }
        
        const res = await fetch(req_url);
        const data = await res.json();

        return data;
    }
}

module.exports = { SubjectsAPI };
