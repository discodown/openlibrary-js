const fetch = require("node-fetch");

class SearchInsideAPI {
    /** Implements the Search Inside API for searching inside an item that is hosted
     * on archive.org.
     * @param {function} callback - Optional callback function to wrap results in.
     */
    constructor(callback = null) {
        this.callback = callback;
    }

    /**
     * Search inside an item.
     * @param {string} hostname - The name of the archive.org host where the item is located.
     * @param {string} id - The archive.org item ID.
     * @param {string} doc - Document name, most times this is the same as the item_id.
     * @param {string} path - Path to the item on the host.
     * @param {string} q - The phrase to search for.
     * @returns {Promise} Promise object containing JSON representation of the search result.
     */
    async searchInside(hostname, id, doc, path, q) {
        let req_url = `https://${hostname}/fulltext/inside.php?item_id=${id}&doc=${doc}&path=${path}&q=${q}`;

        if (this.callback != null) {
            req_url = req_url + `&callback=${this.callback}`;
        }

        const res = await fetch(req_url);
        const data = await res.json();

        return data;
    }

    /**
     * Utility method for getting the name of a host where an item is located on archive.org.
     * @param {string} id - archive.org item ID.
     * @returns {string} Name of the host where the item is located.
     */
    async locateItem(id) {
        const req_url = `https://archive.org/metadata/${id}`;

        const res = await fetch(req_url);
        const data = await res.json();

        const hostname = data["d1"];

        return hostname;
    }

    /**
     * Utility method for getting all of an item's archive.org metadata.
     * @param {string} id - archive.org item ID.
     * @returns {Promise} Promise object containing JSON representation of the item's metadata.
     */
    async getMetadata(id) {
        const req_url = `https://archive.org/metadata/${id}`;

        const res = await fetch(req_url);
        const data = await res.json();

        return data;
    }
}

module.exports = { SearchInsideAPI };
