const fetch = require("node-fetch");

class CoversAPI {
    /**
     * Implements Covers API for fetching book covers by Cover ID, ISBN, etc.
     * @constructor
     * @param {Boolean} notFound - Whether to return a blank image if cover can't be found. Default: true
     */
    constructor(notFound = true) {
        this.BASE_API_URL = "https://covers.openlibrary.org/";
        this.default = notFound;
    }

    /**
     * Get blob of the requested cover.
     * @param {String} key - Type of ID used to get the cover. Can be any one of ISBN, OCLC, LCCN, OLID and ID.
     * @param {String} value - The ID used to get the cover.
     * @param {String} size - Size to return the cover in. Can be S, M, or L. Default: M
     * @returns {Promise} Promise object containing blob representation of the cover.
     */
    async getCoverImageBlob(key, value, size = 'M') {
        const req_url = this.BASE_API_URL + `b/${key}/${value}-${size}.jpg?default=${this.default}`;

        const res = await fetch(req_url);
        const blob = await res.blob();

        return blob;
    }

    /**
     * Get API information about the requested cover.
     * @param {String} key - Type of ID used to get the cover. Can be any one of ISBN, OCLC, LCCN, OLID and ID.
     * @param {String} value - The ID used to get the cover.
     * @param {String} size - Size to return the cover in. Can be S, M, or L. Default: M
     * @returns {Promise} Promise object containing JSON representation of the cover data.
     */
    async getCoverInfo(key, value, size = 'M') {
        const req_url = this.BASE_API_URL + `b/${key}/${value}-${size}.json?default=${this.default}`;

        const res = await fetch(req_url);
        const data = await res.json();

        return data;
    }

    /**
     * Get blob of an book's author photo.
     * @param {String} key - Type of ID used to get the cover. Can be any one of ISBN, OCLC, LCCN, OLID and ID.
     * @param {String} value - The ID used to get the cover.
     * @param {String} size - Size to return the cover in. Can be S, M, or L. Default: M
     * @returns {Promise} Promise object containing blob representation of the author photo.
     */
    async getAuthorPhoto(key, value, size = 'M') {
        const req_url = this.BASE_API_URL + `a/${key}/${value}-${size}.jpg?default=${this.default}`;

        const res = await fetch(req_url);
        const blob = await res.blob();

        return blob;
    }
}

module.exports = { CoversAPI };
