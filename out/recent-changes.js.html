const fetch = require("node-fetch");

class RecentChangesAPI {
    /**
     * Implements the Recent Changes API for fetching the most recent changes made 
     * to Open Library.
     * @constuctor
     * @param {int} limit - Maximum number of results to return. Default: 100
     */
    constructor(limit = 100) {
        this.BASE_API_URL = "http://openlibrary.org/recentchanges";
        this.limit = limit;
    }

    /**
     * Get information about recent changes.
     * @param {Object} modifers - Object containing optional "date" and "kind" modifiers. Default: null for date and kind.
     * @param {Boolean} bot - Specify "true" to only return changes made by bots, "false" for human changes only. Default: null (all changes)
     * @returns {Promise} Promise object containing JSON representation of changes.
     */
    async getRecentChanges(modifers = {"date" : null, "kind" : null}, bot = null) {
        let req_url = this.BASE_API_URL;

        if (modifers["date"] != null) {
            req_url = req_url + `/${modifers["date"]}`;
        }
        if (modifers["kind"] != null) {
            req_url = req_url + `/${modifers["kind"]}`;
        }
        req_url = req_url + `.json?limit=${this.limit}`;
        
        if (bot != null) {
            req_url = req_url + `&bot=${bot}`;
        }
        
        const res = await fetch(req_url);
        const data = await res.json();

        return data;
    }
}

module.exports = { RecentChangesAPI };
