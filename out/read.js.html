const fetch = require("node-fetch");

const idTypes = ["isbn", "lccn", "oclc", "olid"];

class LibraryID {
    /**
     * Object representing a library ID to be used in a request.
     * @constructor
     * @param {*} id - The ID number portion of the library ID.
     * @param {*} idType - The ID type (ISBN, etc.) portion of the library ID.
     */
    constructor(id, idType) {
        this.id = id;
        this.idType = idType;
    }
    /**
     * Return a string representation of the object.
     * @returns {string} String representation of the LibraryID object.
     */
    toString() {
        return `${this.idType}:${this.id}`;
    }
}

class Request {
    /**
     * Object repesenting a request for the Read API. A request consists of a list
     * of LibraryID objects.
     * @constructor
     * @param {LibraryID[]} ids - List of LibraryID objects. Default: empty list.
     */
    constructor(ids = []) {
        this.idList = ids;
    }

    /**
     * Return a string representation of the object.
     * @returns {string} String representation of the Request object.
     */
    toString() {
        let s = "";
        for (const id of this.idList) {
            s = s + id.toString() + ";";
        }

        s = s.substring(0, s.length-1);
        
        return s;
    }
    
    /**
     * Add a LibraryID object to the request.
     * @param {LibraryID} id - LibraryID object to add.
     */
    addLibraryId(id) {
        this.idList.push(id);
    }

    /**
     * Create a new LibraryID object and add it to the request.
     * @param {*} type - The ID type portion of the LibraryID.
     * @param {*} id - The ID portion of the LibraryID.
     */
    addId(type, id) {
        this.idList.push(new LibraryID(type, id));
    }
}

class ReadAPI {
    /**
     * Implements the Read API for getting readable and/or borrowable books. Supports
     * single and multiple requests.
     * @contructors
     * @param {function} callback - Optional callback function to wrap results in. Default: null
     */
    constructor(callback = null) {
        this.BASE_API_URL = "http://openlibrary.org/api/volumes/brief/";
        this.callback = callback;
    }

    /**
     * Get items using a single request.
     * @param {string} id - The requested item's ID.
     * @param {string} idType - The type of ID used to make the request (ISBN, etc).
     * @returns {Promise} Promise object representing the requested item.
     */
    async getSingleRequest(id, idType) {
        let req_url = this.BASE_API_URL + `${idType}/${id}.json`;

        if (this.callback != null) {
            req_url = req_url.concat(`?callback=${this.callback}`);
        }

        const res = await fetch(req_url);
        const data = await res.json();

        return data;
    }

    /**
     * Get items using a multiple requests.
     * @param {Request[]} - List of Request objects.
     * @returns {Promise} Promise object representing the requested items.
     */
    async getMultiRequest(requests) {
        let req = "";

        for (const r of requests) {
            req = req + r.toString() + "|";
        }

        req = req.substring(0, req.length-1);

        let req_url = this.BASE_API_URL + `/json/${req}`;

        if (this.callback != null) {
            req_url = req_url.concat(`?callback=${this.callback}`);
        }

        const res = await fetch(req_url);
        const data = await res.json();

        return data;
    }
}

module.exports = { LibraryID, Request, ReadAPI };
