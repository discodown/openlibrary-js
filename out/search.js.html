class QueryField {
    /**
     * Object representing a query of the form key=value to use when searching.
     * E.g. to search for an author named tolkien, query parameter would be author=tolkien.
     * @param {string} value - Value of the query field.
     * @param {string} key - Query field key (author, title, etc.).
     */
    constructor(value, key = 'q') {
        this.key = key;
        this.value = value;
    }

    /**
     * Return string reprentation of the QueryField object.
     * @returns {string} String reprentation of the QueryField object.
     */
    toString() {
        return `${this.key}=${this.value}`;
    }
}

class SearchAPI {
    /**
     * Implements the Search API for searching for items.
     * @constructor
     * @param {string} sort - Sorting method for results. Default is to sort by relevance. Default: 'default'  
     * @param {string[]} fields - List of fields to return in the results. Defaults to all fields. Default: '*'
     * @param {int} limit - Maximum number of results to return. Default: 50
     */
    constructor(sort = 'default', fields = ['*'], limit = 50) {
        this.BASE_API_URL = "https://openlibrary.org/search";
        if (sort == 'default') {
            this.sort = "";
        }
        else {
            this.sort = sort;
        }
        this.fields = fields;
        this.limit = limit;
    }

    /**
     * Query the search API.
     * @param {QueryField} q - The query to use for the search.
     * @returns {Promise} Promise object containing the search results.
     */
    async search(q) {
        const req_url = this.BASE_API_URL + `.json?${q}&fields=${this.fields.toString()}&sort=${this.sort}&limit=${this.limit}`;
        const res = await fetch(req_url);
        const data = await res.json();

        return data;
    }
    /**
     * Query the author search API.
     * @param {QueryField} q - The query to use for the search.
     * @returns {Promise} Promise object containing the search results.
     */
    async searchAuthors(q) {
        const req_url = this.BASE_API_URL + `/authors.json?${q}&fields=${this.fields.toString()}&sort=${this.sort}&limit=${this.limit}`;
        
        const res = await fetch(req_url);
        const data = await res.json();

        return data;
    }
}
